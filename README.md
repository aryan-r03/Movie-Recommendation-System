<div align="center">

### Content-Based Filtering Algorithm

</div>

The recommendation system uses a **three-step content-based filtering approach**:

### **Step 1: Feature Extraction with TF-IDF**

**What is TF-IDF?**
- **TF (Term Frequency)**: How often a term appears in a document
- **IDF (Inverse Document Frequency)**: How unique/rare a term is across all documents
- **TF-IDF Score**: TF Ã— IDF = Importance of term in document

**Implementation:**
```python
from sklearn.feature_extraction.text import TfidfVectorizer

# Combine movie features
def combine_features(row):
    return f"{row['genres']} {row['overview']} {row['director']}"

movies['combined_features'] = movies.apply(combine_features, axis=1)

# Create TF-IDF vectorizer
vectorizer = TfidfVectorizer(
    max_features=5000,      # Top 5000 most important terms
    ngram_range=(1, 2),     # Unigrams and bigrams
    stop_words='english'    # Remove common English words
)

# Transform text to feature vectors
tfidf_matrix = vectorizer.fit_transform(movies['combined_features'])
```

### **Step 2: Cosine Similarity Calculation**

**What is Cosine Similarity?**
- Measures the angle between two vectors
- Range: 0.0 (completely different) to 1.0 (identical)
- Independent of magnitude, focuses on direction

**Formula:**
```
similarity = cos(Î¸) = (A Â· B) / (||A|| Ã— ||B||)

Where:
- A Â· B = dot product of vectors A and B
- ||A|| = magnitude of vector A
- ||B|| = magnitude of vector B
```

**Implementation:**
```python
from sklearn.metrics.pairwise import cosine_similarity

# Calculate pairwise cosine similarity
cosine_sim = cosine_similarity(tfidf_matrix, tfidf_matrix)

# Result: NxN matrix where N = number of movies
# cosine_sim[i][j] = similarity between movie i and movie j
```

### **Step 3: Recommendation Generation**

**Algorithm:**
```python
def get_recommendations(title, n=5):
    # 1. Find movie index
    idx = movies[movies['title'] == title].index[0]
    
    # 2. Get similarity scores for this movie
    sim_scores = list(enumerate(cosine_sim[idx]))
    
    # 3. Sort by similarity (descending)
    sim_scores = sorted(sim_scores, key=lambda x: x[1], reverse=True)
    
    # 4. Get top N (excluding the movie itself)
    sim_scores = sim_scores[1:n+1]
    
    # 5. Get movie indices
    movie_indices = [i[0] for i in sim_scores]
    
    # 6. Return recommended movies with scores
    recommendations = movies.iloc[movie_indices].copy()
    recommendations['similarity_score'] = [i[1] for i in sim_scores]
    
    return recommendations
```

import pandas as pd
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity
import numpy as np

class MovieRecommender:
    def __init__(self, data_path='../data/movies.csv'):
        """Initialize recommender"""
        # Load dataset
        self.movies = pd.read_csv(data_path)
        
        # Combine features
        self.movies['combined_features'] = self.movies.apply(
            lambda row: f"{row['genres']} {row['overview']} {row['director']}",
            axis=1
        )
        
        # Create TF-IDF vectors
        self.vectorizer = TfidfVectorizer(
            max_features=5000,
            ngram_range=(1, 2),
            stop_words='english'
        )
        
        self.tfidf_matrix = self.vectorizer.fit_transform(
            self.movies['combined_features']
        )
        
        # Calculate cosine similarity
        self.cosine_sim = cosine_similarity(
            self.tfidf_matrix,
            self.tfidf_matrix
        )
        
        # Create title to index mapping
        self.indices = pd.Series(
            self.movies.index,
            index=self.movies['title']
        )
    
    def get_recommendations(self, title, n=5):
        """Get N movie recommendations"""
        # Get movie index
        idx = self.indices[title]
        
        # Get similarity scores
        sim_scores = list(enumerate(self.cosine_sim[idx]))
        
        # Sort by similarity
        sim_scores = sorted(sim_scores, key=lambda x: x[1], reverse=True)
        
        # Get top N (excluding self)
        sim_scores = sim_scores[1:n+1]
        
        # Get movie indices
        movie_indices = [i[0] for i in sim_scores]
        
        # Return recommendations
        recommendations = self.movies.iloc[movie_indices].copy()
        recommendations['similarity_score'] = [i[1] for i in sim_scores]
        
        return recommendations.to_dict('records')
    
    def get_top_rated(self, n=10, min_rating=7.0):
        """Get top-rated movies"""
        top_movies = self.movies[
            self.movies['rating'] >= min_rating
        ].nlargest(n, 'rating')
        
        return top_movies.to_dict('records')
    
    def search_movies(self, query_params):
        """Search movies by criteria"""
        query = query_params.get('query', '')
        filter_by = query_params.get('filter_by', 'title')
        
        # Filter movies
        if filter_by == 'title':
            results = self.movies[
                self.movies['title'].str.contains(query, case=False)
            ]
        elif filter_by == 'genre':
            results = self.movies[
                self.movies['genres'].str.contains(query, case=False)
            ]
        # ... more filters
        
        return results.to_dict('records')
    
    def get_statistics(self):
        """Get system statistics"""
        return {
            'total_movies': len(self.movies),
            'avg_rating': self.movies['rating'].mean(),
            'year_range': {
                'min': int(self.movies['year'].min()),
                'max': int(self.movies['year'].max())
            }
            # ... more stats
        }

from sklearn.neighbors import NearestNeighbors

class HybridRecommender(MovieRecommender):
    def __init__(self, data_path, ratings_path=None):
        super().__init__(data_path)
        
        if ratings_path:
            # Load user ratings
            self.ratings = pd.read_csv(ratings_path)
            
            # Create user-item matrix
            self.user_item_matrix = self.ratings.pivot(
                index='user_id',
                columns='movie_id',
                values='rating'
            ).fillna(0)
            
            # Train KNN model
            self.knn_model = NearestNeighbors(
                metric='cosine',
                algorithm='brute'
            )
            self.knn_model.fit(self.user_item_matrix.T)
    
    def get_hybrid_recommendations(self, title, user_id=None, n=5):
        """Combine content-based and collaborative filtering"""
        # Get content-based recommendations
        content_recs = self.get_recommendations(title, n*2)
        
        if user_id and hasattr(self, 'knn_model'):
            # Get collaborative filtering recommendations
            movie_id = self.movie_to_id[title]
            distances, indices = self.knn_model.kneighbors(
                self.user_item_matrix.T.iloc[movie_id, :].values.reshape(1, -1),
                n_neighbors=n*2
            )
            
            collab_recs = [self.id_to_movie[i] for i in indices[0]]
            
            # Combine recommendations (weighted)
            # 70% content-based, 30% collaborative
            combined = self.merge_recommendations(
                content_recs,
                collab_recs,
                weights=[0.7, 0.3]
            )
            
            return combined[:n]
        
        return content_recs[:n]

class MovieRecommenderWithTrailers(MovieRecommender):
    def get_trailer_url(self, movie_title, year):
        """Get YouTube trailer URL"""
        search_query = f"{movie_title} {year} official trailer"
        
        videos_search = VideosSearch(search_query, limit=1)
        results = videos_search.result()
        
        if results['result']:
            return results['result'][0]['link']
        return None
    
    def get_recommendations_with_trailers(self, title, n=5):
        """Get recommendations with trailer links"""
        recommendations = self.get_recommendations(title, n)
        
        for movie in recommendations:
            trailer_url = self.get_trailer_url(
                movie['title'],
                movie['year']
            )
            movie['trailer_url'] = trailer_url
        
        return recommendations
